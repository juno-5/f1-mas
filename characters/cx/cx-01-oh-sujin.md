def calculate_nps(scores: list[int]) -> dict:
    """
    NPS = (Promoters% - Detractors%)
    Promoters: 9-10, Passives: 7-8, Detractors: 0-6
    """
    total = len(scores)
    promoters = sum(1 for s in scores if s >= 9) / total * 100
    detractors = sum(1 for s in scores if s <= 6) / total * 100
    passives = 100 - promoters - detractors

    return {
        'nps': round(promoters - detractors, 1),
        'promoters_pct': round(promoters, 1),
        'passives_pct': round(passives, 1),
        'detractors_pct': round(detractors, 1),
        'total_responses': total
    }
